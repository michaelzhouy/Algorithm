# 0074-搜索二维矩阵

## 说明
编写一个高效的算法来判断 m x n 矩阵中，是否存在一个目标值。该矩阵具有如下特性：
- 每行中的整数从左到右按升序排列。
- 每行的第一个整数大于前一行的最后一个整数。

## 示例
```
输入：matrix = [[1,3,5,7],[10,11,16,20],[23,30,34,60]], target = 3
输出：true

输入：matrix = [[1,3,5,7],[10,11,16,20],[23,30,34,60]], target = 13
输出：false
```

class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        m, n = len(matrix), len(matrix[0])
        lastCol = [matrix[i][n - 1] for i in range(m)]
        # left, right = 0, m - 1
        # while left > right:
        def findCol(left, right, lastCol, target):
            if target == lastCol[left]:
                return True
            elif target < lastCol[left]:
                return left
            while left + 1 != right:
                mid = (left + right) // 2
                if target < lastCol[mid]:
                    right = mid
                elif target > lastCol[mid]:
                    left = mid
                else:
                    return True
            return right
        
        colIndex = findCol(0, m - 1, lastCol, target)
        print(colIndex)
        row = matrix[colIndex][:]
        print(row)
        res = findCol(0, n - 1, row, target)
        return res == True
